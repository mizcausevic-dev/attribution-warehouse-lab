warehouse contracts

Warehouse contracts matter as much as the credit model.

If the staging grain, fact grain, and mart grain are fuzzy, the attribution model is just decoration. This page keeps the actual warehouse contract visible.

stg_touchpoints

one row per attributable touchpoint

journey_id, touch_timestamp, channel, campaign, stage, touch_cost

fct_attribution_journeys

one row per converted account journey

journey_id, account, segment, conversion_type, pipeline_dollars, days_to_convert

mart_channel_credit

one row per channel per model

model_name, channel, allocated_pipeline, allocated_cost, roi_ratio

-- fct_attribution_journeys.sql
with touchpoints as (
    select * from {{ ref('stg_touchpoints') }}
),
journey_rollup as (
    select
        journey_id,
        any_value(account_name) as account_name,
        any_value(segment) as segment,
        min(touch_timestamp) as first_touch_at,
        max(touch_timestamp) as last_touch_at,
        count(*) as touch_count
    from touchpoints
    group by 1
)
select
    j.journey_id,
    j.account_name,
    j.segment,
    j.first_touch_at,
    j.last_touch_at,
    j.touch_count,
    c.conversion_type,
    c.pipeline_dollars,
    datediff(day, j.first_touch_at, c.converted_at) as days_to_convert
from journey_rollup j
join {{ source('revenue', 'conversions') }} c
  on j.journey_id = c.journey_id


-- mart_channel_credit.sql
with journeys as (
    select * from {{ ref('fct_attribution_journeys') }}
),
touchpoints as (
    select * from {{ ref('stg_touchpoints') }}
),
weighted_credit as (
    select
        t.journey_id,
        t.channel,
        case
            when t.stage = 'first_touch' then 0.40
            when t.stage = 'last_touch' then 0.40
            else 0.20 / nullif(j.touch_count - 2, 0)
        end as credit_share,
        t.touch_cost
    from touchpoints t
    join journeys j
      on t.journey_id = j.journey_id
)
select
    'position_weighted' as model_name,
    channel,
    sum(j.pipeline_dollars * w.credit_share) as allocated_pipeline,
    sum(w.touch_cost) as allocated_cost,
    sum(j.pipeline_dollars * w.credit_share) / nullif(sum(w.touch_cost), 0) as roi_ratio
from weighted_credit w
join journeys j
  on w.journey_id = j.journey_id
group by 1, 2


-- stg_touchpoints.sql
select
    journey_id,
    touch_timestamp,
    account_name,
    segment,
    channel,
    campaign,
    stage,
    touch_cost
from {{ source('marketing', 'touchpoints') }}
where touch_timestamp >= dateadd(day, -120, current_timestamp)
